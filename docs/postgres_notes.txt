DROP DATABASE IF EXISTS auth;
DROP DATABASE IF EXISTS broker;
DROP DATABASE IF EXISTS chinook;

CREATE DATABASE auth;
CREATE DATABASE broker;
CREATE DATABASE chinook;

CREATE USER auth WITH PASSWORD 'xxx';
CREATE USER broker WITH PASSWORD 'xxx';
CREATE USER chinook WITH PASSWORD 'xxx';

GRANT ALL PRIVILEGES ON SCHEMA public TO auth;
GRANT CREATE ON DATABASE broker TO broker;
GRANT ALL PRIVILEGES ON SCHEMA public TO chinook;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO chinook;
